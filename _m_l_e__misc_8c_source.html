197 static void radix_sort_recursive(uint32_t *data, int count, int ndim,
198  int *order, int *tmp_order, int sdim, int sbit,
199  int lo, int hi)
200 // Hierarchical radix sort to get Z-order of particles.
201 // This function is static not to be visible outside this module.
202 {
203  int i, lo_last = lo, hi_last = hi;
204  uint32_t *sdata = data+sdim*count;
205  uint32_t check = 1 << sbit;
206  for(i = lo; i <= hi; i++)
207  {
208  if((sdata[order[i]] & check) == 0)
209  {
210  tmp_order[lo_last] = order[i];
211  lo_last++;
212  }
213  else
214  {
215  tmp_order[hi_last] = order[i];
216  hi_last--;
217  }
218  }
219  for(i = lo; i <= hi; i++)
220  order[i] = tmp_order[i];
221  if(sdim > 0)
222  {
223  if(lo_last-lo > 1)
224  radix_sort_recursive(data, count, ndim, order, tmp_order, sdim-1,
225  sbit, lo, lo_last-1);
226  if(hi-hi_last > 1)
227  radix_sort_recursive(data, count, ndim, order, tmp_order, sdim-1,
228  sbit, hi_last+1, hi);
229  }
230  else if(sbit > 0)
231  {
232  if(lo_last-lo > 1)
233  radix_sort_recursive(data, count, ndim, order, tmp_order, ndim-1,
234  sbit-1, lo, lo_last-1);
235  if(hi-hi_last > 1)
236  radix_sort_recursive(data, count, ndim, order, tmp_order, ndim-1,
237  sbit-1, hi_last+1, hi);
238  }
239 }
